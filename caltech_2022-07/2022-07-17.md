```
tee mysql.yaml 0<<EOF
apiVersion: v1
kind: Pod
metadata:
  labels:
    app: mysql
  name: po-mysql
spec:
  containers:
  - image: mysql:5.6
    name: container-mysql
    env:
    - name: MYSQL_ROOT_PASSWORD
      value: simplilearn
    - name: MYSQL_DATABASE
      value: database1
EOF
```
```
kubectl create -f mysql.yaml
```
```
tee wordpress.yaml 0<<EOF
apiVersion: apps/v1
kind: Deployment
metadata:
  labels:
    app: wordpress
  name: deploy-wordpress
spec:
  replicas: 1
  selector:
    matchLabels:
      app: wordpress
  template:
    metadata:
      labels:
        app: wordpress
    spec:
      containers:
      - image: docker.io/library/wordpress:latest
        name: container-wordpress
        env:
        - name: WORDPRESS_DB_HOST
          value: mysql
        - name: WORDPRESS_DB_PASSWORD
          value: simplilearn
        - name: WORDPRESS_DB_USER
          value: root
        - name: WORDPRESS_DB_NAME
          value: database1
EOF
```
```
kubectl create -f wordpress.yaml
kubectl expose po po-mysql --port 3306 --protocol TCP --target-port 3306 --name mysql --type ClusterIP
kubectl expose deploy deploy-wordpress --port 80 --protocol TCP --target-port 80 --name wordpress --type NodePort
```
