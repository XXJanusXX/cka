# Lesson 4 Demo 11
## Understanding Config Maps and Secrets
```
dir=${HOME}/Lesson_4/Demo_11
mkdir --parents ${dir}
cd ${dir}
```
```
tee cm.yaml 0<<EOF
kind: ConfigMap 
apiVersion: v1 
metadata:
  name: example-configmap 
data:
  # Configuration values can be set as key-value properties
  database: mongodb
  database_uri: mongodb://localhost:27017
EOF
```
```
kubectl create -f cm.yaml
```
```
tee po.yaml 0<<EOF
kind: Pod 
apiVersion: v1 
metadata:
  name: pod-env-var 
spec:
  containers:
    - name: env-var-configmap
      image: nginx:1.7.9 
      envFrom:
        - configMapRef:
            name: example-configmap
EOF
```
```
kubectl create -f po.yaml
kubectl exec pod/pod-env-var -- printenv | grep -E "database=|database_uri="
```
```
tee po-2.yaml 0<<EOF
kind: Pod 
apiVersion: v1 
metadata:
  name: pod-env12
spec:
  containers:
    - name: env-var-configmap
      image: nginx:1.7.9 
      env:
        - name: testenv
          valueFrom: 
           configMapKeyRef:
             name: example-configmap
             key: database
EOF
```
```
kubectl create -f po-2.yaml
kubectl exec pod/pod-env12 -- printenv | grep -E "database=|database_uri="
kubectl exec pod/pod-env12 -- printenv | grep -E testenv=
```

# Lesson 4 Demo 6
## Using Init Containers
```
dir=${HOME}/Lesson_4/Demo_6
mkdir --parents ${dir}
cd ${dir}
```
```
tee init-container.yaml 0<<EOF
apiVersion: v1
kind: Pod
metadata:
  name: myapp-pod
  labels:
    app: myapp
spec:
  containers:
  - name: myapp-container
    image: busybox:1.28
    command: ['sh', '-c', 'echo The app is running! && sleep 3600']
  initContainers:
  - name: init-myservice
    image: busybox:1.28
    command: ['sh', '-c', "until nslookup myservice.\$(cat /var/run/secrets/kubernetes.io/serviceaccount/namespace).svc.cluster.local; do echo waiting for myservice; sleep 2; done"]
EOF
```
```
kubectl create -f init-container.yaml
watch kubectl get po/myapp-pod
```
```
kubectl logs po/myapp-pod -c init-myservice --follow
```
```
tee init-svc.yaml 0<<EOF
kind: Service
apiVersion: v1
metadata:
  name: myservice
spec:
  ports:
  - protocol: TCP
    port: 80
    targetPort: 9376
EOF
```
```
kubectl create -f init-svc.yaml
kubectl logs po/myapp-pod -c init-myservice
kubectl logs myapp-pod
```
